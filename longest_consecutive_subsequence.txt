Given an array of positive integers. 
Find the length of the longest sub-sequence such that elements in the subsequence are consecutive integers, the consecutive numbers can be in any order.

int findLongestConseqSubseq(int arr[], int N)
    {
      map<int,int> trace;
      for(int i=0;i<N;i++){
          trace[arr[i]]++;
      }
      int res=1;
      int count;
      int initial=-1;
      for(auto &it:trace){
          if(initial==-1 || initial+1!=it.first){ //first element of sequence
              initial=it.first;
              count=1;
          }
          else{
             count++;
             initial=it.first;
          }
          res=max(res,count);
      }
      
      return res;
    }
