 Given an integer N denoting the Length of a line segment. You need to cut the line segment in such a way 
 that the cut length of a line segment each time is either x , y or z. Here x, y, and z are integers.
After performing all the cut operations, your total number of cut segments must be maximum.
 
 //Function to find the maximum number of cuts.
    int maximizeTheCuts(int n, int x, int y, int z)
    {
        //Your code here
        vector<int> dp(n+1,-1); // 0 to n
        dp[0]=0;
        for(int i=0;i<=n;i++){ //only if we start from 0 initially p,q,r will not be -1.
            if(dp[i]==-1)
                continue;
                
            if(i+x<=n){
                dp[i+x]=max(dp[i+x],1+dp[i]);
            }    
            if(i+y<=n){
                dp[i+y]=max(dp[i+y],1+dp[i]);
            }
            if(i+z<=n){
                dp[i+z]=max(dp[i+z],1+dp[i]);
            }
        }
        if(dp[n]==-1)
            return 0;
        return dp[n];
    }
