Selection Sort

#include <iostream>
using namespace std;
int main()
{
    int arr[]={3,6,4,2,53,3,21,56,78,2,42,244, 234,2442,23323, 13,355454};
    int n=sizeof arr/sizeof(int);
    for(int i=0;i<n;i++){
        int min=arr[i],index=i;
        for(int j=i+1;j<n;j++){
            if(arr[j]<min){
                min=arr[j];
                index=j;
            }
        }
        int temp=arr[i];
        arr[i]=arr[index];
        arr[index]=temp;
    }
    for(int i=0;i<n;i++){
        cout<<arr[i]<<" ";
    }
    return 0;
}
