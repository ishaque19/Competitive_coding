#include <iostream>
using namespace std;
int main()
{
    int arr[]={3,6,4,2,53,3,21,56,78,2,42,244, 2442,23323, 13,355454};
    int n=sizeof arr/sizeof(int);
    int eof=0;
    for(int i=1;i<n;i++){
        for(int j=0;j<=eof;){
            if(arr[j]<=arr[i]){
                j++;
            }
            else{
                int temp=arr[i],k=i;
                while(k>j){
                    arr[k]=arr[k-1];
                    k--;
                }
                arr[j]=temp;
                break;
           }
        }
        eof++;
    }
    for(int i=0;i<n;i++){
        cout<<arr[i]<<" ";
    }
    return 0;
}
