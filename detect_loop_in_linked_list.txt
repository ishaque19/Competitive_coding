 bool detectLoop(Node* head)
    {
        Node *p=head;
        Node *q=head;
        int flag=0;
        while(q->next && q->next->next){
            p=p->next;
            q=q->next->next;
            if(p==q){
                flag=1;break;
            }
        }
        
        if(flag)
            return true;
        else
            return false;
    }
