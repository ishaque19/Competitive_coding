Check if two arrays are equal or not.
 class Solution{
    public:

      unordered_map<ll,int> track;
    bool check(vector<ll> A, vector<ll> B, int N) {
        for(int i=0;i<N;i++){
            track[A[i]]++;
            track[B[i]]--;
        }
        
        for(auto &it:track){
            if(it.second!=0)
                return false;
        }
        
        return true;
    }
};
